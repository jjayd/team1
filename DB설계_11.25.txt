<코드>

CREATE TABLE MESSAGE(
 MNO VARCHAR(10) NOT NULL,
 Seq SmallINT NOT NULL,
 UNO VARCHAR(10) NOT NULL,
 CNO VARCHAR(10) NOT NULL,
 RNO VARCHAR(10) NOT NULL,
 Content VARCHAR(256) NOT NULL,
 MessageTime DATE NOT NULL,
 PRIMARY KEY(MNO)
);

CREATE TABLE USER(
 UNO INT NOT NULL,
 Uname VARCHAR(10) CHARACTER SET UTF8 NOT NULL,
 Gender VARCHAR(1) NOT NULL,
 Udate DATE NOT NULL,
 ID VARCHAR(30) NOT NULL,
 Password VARCHAR(30) NOT NULL,
 PhoneNumber VARCHAR(30),
 E_mail VARCHAR(256),
 Depart VARCHAR(256) CHARACTER SET UTF8,
 Grade VARCHAR(256),
 PRIMARY KEY(UNO),
 UNIQUE(ID)
);

CREATE TABLE USER_FIELD(
 UNO INT NOT NULL,
 FIELD VARCHAR(20) NOT NULL,
 PRIMARY KEY(UNO,FIELD)
);

CREATE TABLE ROOM(
 CNO VARCHAR(1) NOT NULL,
 RNO VARCHAR(10) NOT NULL,
 MaxUser TINYINT NOT NULL,
 CurUser TINYINT NOT NULL,
 UNO VARCHAR(10) NOT NULL,
 IsExist BOOL NOT NULL,
 PRIMARY KEY(CNO, RNO)
);

CREATE TABLE RECRUIT(
 CNO VARCHAR(1) NOT NULL,
 RNO INT NOT NULL,
 Field VARCHAR(20) CHARACTER SET UTF8 NOT NULL,
 Title VARCHAR(50) CHARACTER SET UTF8 NOT NULL,
 Contents TEXT CHARACTER SET UTF8 NOT NULL,
 FinishDate DATE NOT NULL,
 URL VARCHAR(100) NOT NULL,
 PostDate DATE NOT NULL,
 PRIMARY KEY(CNO, RNO)
);

CREATE TABLE CONTEST(
 CNO VARCHAR(1) NOT NULL,
 RNO INT NOT NULL,
 Field VARCHAR(20) CHARACTER SET UTF8 NOT NULL,
 Topic VARCHAR(20) CHARACTER SET UTF8 NOT NULL,
 Reward VARCHAR(20) CHARACTER SET UTF8 NOT NULL,
 Organizer VARCHAR(20) CHARACTER SET UTF8 NOT NULL,
 Title VARCHAR(50) CHARACTER SET UTF8 NOT NULL,
 Contents TEXT CHARACTER SET UTF8 NOT NULL,
 FinishDate DATE NOT NULL,
 URL VARCHAR(100) NOT NULL,
 PostDate DATE NOT NULL,
 PRIMARY KEY(CNO, RNO)
);

CREATE TABLE OVER_RECRUIT(
 CNO VARCHAR(1) NOT NULL,
 RNO INT NOT NULL,
 ORIGIN VARCHAR(20) CHARACTER SET UTF8 NOT NULL,
 URL VARCHAR(100) NOT NULL,
 PRIMARY KEY(CNO, RNO)
);
